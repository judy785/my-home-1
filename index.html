import streamlit as st import pandas as pd import FinanceDataReader as fdr import plotly.express as px import plotly.graph_objects as go # 1. 앱 페이지 설정 st.set_page_config(page_title="AI 공급망 리서치 대시보드", layout="wide") # 2. 데이터 수집 함수 (한국 거래소 종목 및 주가) @st.cache_data def get_stock_data(): # 코스피, 코스닥 종목 리스트 불러오기 df_krx = fdr.StockListing('KRX') return df_krx # 3. 분석 대상 기업 데이터 (리서치 기반 수동 입력 및 매핑) target_companies = { "필옵틱스": {"code": "161580", "sector": "유리기판", "ai_rev": 35, "target_per": 22}, "GST": {"code": "083450", "sector": "액침냉각", "ai_rev": 20, "target_per": 35}, "마이크로컨텍솔": {"code": "098120", "sector": "HBM소켓", "ai_rev": 45, "target_per": 28}, "샘씨엔에스": {"code": "252990", "sector": "테스트부품", "ai_rev": 30, "target_per": 32}, "디아이티": {"code": "110990", "sector": "레이저어닐링", "ai_rev": 40, "target_per": 40} } # --- 사이드바: 설정 --- st.sidebar.header("📍 분석 설정") ai_growth_rate = st.sidebar.slider("글로벌 AI Capex 예상 성장률 (%)", 10, 50, 30) # --- 메인 화면 --- st.title("🚀 글로벌 AI 공급망 수혜주 분석") st.markdown(f"**현재 설정:** 향후 5년간 연평균 성장률(CAGR) **{ai_growth_rate}%** 가정") # 4. 실시간 시세 및 밸류에이션 계산 st.subheader("1. 추천 종목 실시간 현황 및 밸류에이션") col1, col2 = st.columns([2, 1]) results = [] for name, info in target_companies.items(): # 실제 주가 데이터 수집 (예시로 현재가만 추출) curr_price = fdr.DataReader(info['code']).iloc[-1]['Close'] results.append({ "종목명": name, "현재가": curr_price, "섹터": info['sector'], "AI 매출비중(%)": info['ai_rev'], "현재 PER": round(10.5, 2), # 실제 API 연동 필요 (ex: FnGuide) "글로벌 Peer PER": info['target_per'] }) df_results = pd.DataFrame(results) col1.dataframe(df_results, use_container_width=True) # 5. 시각화: 글로벌 대비 저평가 매력도 fig = px.bar(df_results, x="종목명", y=["현재 PER", "글로벌 Peer PER"], barmode="group", title="글로벌 Peer 대비 밸류에이션 비교") col2.plotly_chart(fig, use_container_width=True) # 6. 매출 성장 로드맵 (시나리오 분석) st.divider() st.subheader("2. 향후 2년간 매출 성장 로드맵 (Scenario)") selected_stock = st.selectbox("종목을 선택하세요", list(target_companies.keys())) # 가상의 시나리오 계산 로직 base_revenue = 1000 # 기본 매출 (억원) years = ['2025 (E)', '2026 (E)'] bull = [base_revenue * (1 + (ai_growth_rate+10)/100), base_revenue * (1 + (ai_growth_rate+10)/100)**2] base = [base_revenue * (1 + ai_growth_rate/100), base_revenue * (1 + ai_growth_rate/100)**2] bear = [base_revenue * (1 + (ai_growth_rate-10)/100), base_revenue * (1 + (ai_growth_rate-10)/100)**2] fig_scenario = go.Figure() fig_scenario.add_trace(go.Scatter(x=years, y=bull, name='Bull Case', line=dict(color='green', dash='dot'))) fig_scenario.add_trace(go.Scatter(x=years, y=base, name='Base Case', line=dict(color='blue', width=4))) fig_scenario.add_trace(go.Scatter(x=years, y=bear, name='Bear Case', line=dict(color='red', dash='dot'))) st.plotly_chart(fig_scenario, use_container_width=True)
